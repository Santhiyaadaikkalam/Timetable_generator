import os
import sys
from xhtml2pdf import pisa
from pygments import highlight
from pygments.lexers import PythonLexer
from pygments.formatters import HtmlFormatter

# Read the Python file
file_name = "manage.py"
with open(file_name, "r") as f:
    code = f.read()

# Syntax highlighting
formatter = HtmlFormatter(full=True, style="monokai")
highlighted_code = highlight(code, PythonLexer(), formatter)

# Convert to PDF
pdf_file = "manage.pdf"
with open("manage.html", "w") as html_file:
    html_file.write(highlighted_code)

with open("manage.html", "r") as html_file, open(pdf_file, "wb") as output_pdf:
    pisa.CreatePDF(html_file, dest=output_pdf)

print(f"PDF saved as {pdf_file}")
